压缩

执行下面的代码，压缩指定的文件。

const fs = require('fs');
const zlib = require('zlib');
fs.readFile('demo.txt', (err, data) => {
    if (err) return console.error(err);
    zlib.gzip(data, (err, data) => {
        if (err) return console.error(err);
        fs.writeFile('demo.txt.gz', data, err => {
            if (err) return console.error(err);
            console.log('Compression completed.');
        });
    });
});

执行下面的代码，同步压缩文件。

const fs = require('fs');
const zlib = require('zlib');
let file = fs.readFileSync('demo.txt');
let gzip = zlib.gzipSync(file);
fs.writeFileSync('demo.txt.gz', gzip);

执行下面的代码，解压指定的文件。

const fs = require('fs');
const zlib = require('zlib');
fs.readFile('demo.txt.gz', (err, data) => {
    if (err) return console.error(err);
    zlib.gunzip(data, (err, data) => {
        if (err) return console.error(err);
        fs.writeFile('demo.txt', data, err => {
            if (err) return console.error(err);
            console.log('Decompression completed.');
        });
    });
});

执行下面的代码，同步解压文件。

const fs = require('fs');
const zlib = require('zlib');
let file = fs.readFileSync('demo.txt.gz');
let gunzip = zlib.gunzipSync(file);
fs.writeFileSync('demo.txt', gunzip);
