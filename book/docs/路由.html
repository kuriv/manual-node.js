路由

路由

执行下面的代码，将 URL 字符串解析为对象。

const url = require('url');
let string = 'http://www.example.com:9000/home?foo=bar&baz=qux#hash';
let obj = url.parse(string);
console.log(obj);
console.log(obj.protocol);
console.log(obj.host);
console.log(obj.hostname);
console.log(obj.port);
console.log(obj.pathname);
console.log(obj.query);
console.log(obj.hash);

执行下面的代码，将对象转换为 URL 字符串。

const url = require('url');
let obj = {
    protocol: 'http',
    host: 'www.example.com:9000',
    hostname: 'www.example.com',
    port: 9000,
    pathname: '/home',
    query: {
        foo: 'bar',
        baz: 'qux'
    },
    hash: '#hash'
};
console.log(url.format(obj));

也可以执行下面的代码，将 URL 字符串解析为对象。

let url = new URL('http://www.example.com:9000/home?foo=bar&baz=qux#hash');
console.log(url);
console.log(url.protocol);
console.log(url.host);
console.log(url.hostname);
console.log(url.port);
console.log(url.pathname);
console.log(url.searchParams);
console.log(url.hash);

执行下面的代码，处理 URL 查询字符串。

let url = new URL('http://www.example.com:9000/home?foo=bar&baz=qux#hash');
let params = url.searchParams;
console.log(params);
console.log(params.get('foo'));
console.log(params.get('baz'));
params.append('num', '233');
params.delete('foo');
console.log(params);

执行下面的代码，解析相对于基础 URL 的目标 URL 。

const url = require('url');
let string1 = 'http://www.example.com:9000/home';
let string2 = 'foo/bar';
console.log(url.resolve(string1, string2));

执行下面的代码，创建 HTTP 路由。

const http = require('http');
const server = http.createServer((req, res) => res.end('Hello Node.js!'));
server.listen(9000, () => console.log('Server is running.'));

执行下面的代码，获取请求头相关信息。

const http = require('http');
const server = http.createServer((req, res) => {
    res.write(`${req.method}\r\n`);
    res.write(`${req.url}\r\n`);
    res.write(`${req.httpVersion}\r\n`);
    res.write(`${req.headers.host}\r\n`);
    res.write(`${req.headers.connection}\r\n`);
    res.write(`${req.headers.accept}\r\n`);
    res.end('Hello Node.js!');
});
server.listen(9000, () => console.log('Server is running.'));

执行下面的代码，获取请求路径。

const http = require('http');
const url = require('url');
const server = http.createServer((req, res) => {
    let obj = url.parse('http://' + req.headers.host + req.url);
    res.write(`${url.format(obj)}\r\n`);
    res.end('Hello Node.js!');
});
server.listen(9000, () => console.log('Server is running.'));

执行下面的代码，设置响应头相关信息。

const http = require('http');
const server = http.createServer((req, res) => {
    res.setHeader('Content-Type', 'text/html;charset=utf-8');
    res.setHeader('Node', 'Hello World!');
    res.statusCode = 404;
    res.statusMessage = 'OK';
    res.end('Hello Node.js!');
});
server.listen(9000, () => console.log('Server is running.'));

执行下面的代码，通过读取文件内容的方式响应请求。

const http = require('http');
const fs = require('fs');
const server = http.createServer((req, res) => {
    fs.readFile(__dirname + '/demo.html', (err, data) => {
        if (err) {
            res.statusCode = 404;
            res.end('404 Not Found.');
            return;
        }
        res.end(data);
    });
});
server.listen(9000, () => console.log('Server is running.'));

执行下面的代码，处理不同类型的请求。

const http = require('http');
const server = http.createServer((req, res) => {
    switch (req.method) {
        case 'GET':
            res.end('Hello Node.js!');
            break;
        case 'POST':
            res.end('Hello World!');
            break;
        default:
            res.statusCode = 404;
            res.end('404 Not Found.');
            break;
    }
});
server.listen(9000, () => console.log('Server is running.'));
